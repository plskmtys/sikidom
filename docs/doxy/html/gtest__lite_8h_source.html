352template <typename T>
353bool almostEQ(T a, T b) {
354 // eps: ha a relatív, vagy abszolút hiba ettől kisebb, akkor elfogadjuk
355 T eps = 10 * std::numeric_limits<T>::epsilon(); // 10-szer a legkisebb érték
356 if (a == b) return true;
357 if (fabs(a - b) < eps)
358 return true;
359 double aa = fabs(a);
360 double ba = fabs(b);
361 if (aa < ba) {
362 aa = ba;
363 ba = fabs(a);
364 }
365 return (aa - ba) < aa * eps;
366}
